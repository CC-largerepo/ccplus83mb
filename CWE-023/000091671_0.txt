#else

#endif

#ifdef _WIN32
#define OPEN _wopen
#define CLOSE _close
#else
#include <unistd.h>
#define OPEN open
#define CLOSE close
#endif

namespace var1
{



static void func3(wchar_t * data)
{
    {
        int fileDesc;
        fileDesc = OPEN(data, O_RDWR|O_CREAT, S_IREAD|S_IWRITE);
        if (fileDesc != -1)
        {
            CLOSE(fileDesc);
        }
    }
}

static void func4()
{
    wchar_t * data;
    void (*funcPtr) (wchar_t *) = func3;
    wchar_t dataBuffer[FILENAME_MAX] = BASEPATH;
    data = dataBuffer;
    wcscat(data, L"file.txt");
    funcPtr(data);
}

void func5()
{
    func4();
}

} 



using namespace var1; 

int main(int argc, char * argv[])
{
