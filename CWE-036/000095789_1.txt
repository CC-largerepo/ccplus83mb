            if (FILENAME_MAX-dataLen > 1)

                {
                    dataLen = strlen(data);
                    if (dataLen > 0 && data[dataLen-1] == '\n')
                    {
                        data[dataLen-1] = '\0';
                    }
                }
                else
                {
                    printLine("fgets() failed");
                    data[dataLen] = '\0';
                }
            }
        }
    }
    else
    {

        strcat(data, "c:\\temp\\file.txt");
#else
        strcat(data, "/tmp/file.txt");
#endif
    }
    {

        pFile = FOPEN(data, "wb+");
        if (pFile != NULL)
        {
            fclose(pFile);
        }
    }
