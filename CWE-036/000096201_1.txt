namespace GenNS1
{

typedef struct _structType
{
    char * structFirst;
} structType;



void func1(structType myStruct)
{
    char * data = myStruct.structFirst;
    {
        int fileDesc;
        fileDesc = OPEN(data, O_RDWR|O_CREAT, S_IREAD|S_IWRITE);
        if (fileDesc != -1)
        {
            CLOSE(fileDesc);
        }
    }
}
