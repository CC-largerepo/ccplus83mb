namespace GenNS1
{

class GenClass1
{
public:

};



class GenClass2 : public GenClass1
{
public:
    void action(wchar_t * data) const;
};

}



#ifdef _WIN32
#define FOPEN _wfopen
#else
#define FOPEN fopen
#endif

namespace GenNS1
{

void GenClass2::action(wchar_t * data) const
{
    {
        FILE *pFile = NULL;
        pFile = FOPEN(data, L"wb+");
        if (pFile != NULL)
        {
            fclose(pFile);
        }
    }
}

}
