typedef struct var1
{
    char * structFirst;
} var2;



void func1(var2 myStruct)
{
    char * data = myStruct.structFirst;
    {
        HMODULE hModule;
        hModule = LoadLibraryA(data);
        if (hModule != NULL)
        {
            FreeLibrary(hModule);
            printLine("Library loaded and freed successfully");
        }
        else
        {
            printLine("Unable to load library");
        }
    }
}
