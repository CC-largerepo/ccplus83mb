}

{

}

{

    wchar_t dataBuffer[100] = L"";
    data = dataBuffer;
    if(staticReturnsFalse())
    {
        printLine("Benign, fixed string");
    }
    else
    {
        wcscpy(data, L"C:\\Windows\\System32\\winsrv.dll");

        HMODULE hModule;
        hModule = LoadLibraryW(data);

        else
        {
            printLine("Unable to load library");

}
