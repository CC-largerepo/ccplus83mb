#include <wchar.h>

#include <windows.h>



void func1()
{
    wchar_t * data;
    wchar_t dataBuffer[100] = L"";
    data = dataBuffer;
    switch(6)
    {
    case 6:
        wcscpy(data, L"winsrv.dll");
        break;
    default:
        printLine("Benign, fixed string");
        break;
    }
    {
        HMODULE hModule;
        hModule = LoadLibraryW(data);
        if (hModule != NULL)
        {
            FreeLibrary(hModule);

        }
        else
        {
            printLine("Unable to load library");
        }
    }
}



int main(int argc, char * argv[])
