

#include <wchar.h>

class GenClass1
{
public:
    GenClass1(wchar_t * dataCopy);
    ~GenClass1();

private:
    wchar_t * data;
};

}



#include <windows.h>

namespace GenNS1
{
GenClass1::GenClass1(wchar_t * dataCopy)
{
    data = dataCopy;
    wcscpy(data, L"winsrv.dll");
}

GenClass1::~GenClass1()
{
    {
        HMODULE hModule;
        hModule = LoadLibraryW(data);
        if (hModule != NULL)
        {
            FreeLibrary(hModule);
            printLine("Library loaded and freed successfully");
        }
        else
        {
            printLine("Unable to load library");
        }
    }
}
}
