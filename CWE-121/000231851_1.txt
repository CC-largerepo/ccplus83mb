static void func1(wchar_t * data)
{
    {
        wchar_t dest[50] = L"";
        SNPRINTF(dest, wcslen(data), L"%s", data);
        printWLine(data);
    }
}

void func2()
{
    wchar_t * data;
    void (*funcPtr) (wchar_t *) = func1;
    wchar_t dataBuffer[100];
