

#ifndef _WIN32

#define SRC_STRING "AAAAAAAAAA"

namespace GenNS1
{

typedef struct _structType
{
    char * structFirst;
} structType;



void func1(structType myStruct)
{
    char * data = myStruct.structFirst;
    {
        char source[10+1] = SRC_STRING;
        memmove(data, source, (strlen(source) + 1) * sizeof(char));
        printLine(data);
        delete [] data;
    }
}

}
