{

};

{
public:
    void action(wchar_t * data);
};

}



namespace GenNS1
{

void GenClass3::action(wchar_t * data)
{
    {
        wchar_t source[100];
        wmemset(source, L'C', 100-1); 
        source[100-1] = L'\0'; 
        wcsncat(data, source, 100);
        printWLine(data);
        delete [] data;
    }
}

}


namespace GenNS1
{



static void func3()
{

void func4()
{
    func3();
}

}
