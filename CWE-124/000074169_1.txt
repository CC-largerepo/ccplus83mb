static void func1(int data)
{
    {
        int i;
        int buffer[10] = { 0 };
        if (data < 10)
        {
            buffer[data] = 1;
            for(i = 0; i < 10; i++)
            {
                printIntLine(buffer[i]);
            }
        }
        else
        {
            printLine("ERROR: Array index is negative.");
        }
    }
}

void func2()
{
    int data;
    void (*funcPtr) (int) = func1;
    data = -1;

}



int main(int argc, char * argv[])
{
