

#include <wchar.h>

namespace GenNS1
{

typedef struct _structType
{
    char * structFirst;
} structType;



void func2(structType myStruct)
{
    char * data = myStruct.structFirst;
    {
        size_t i, destLen;
        char dest[100];
        memset(dest, 'C', 100-1);
        dest[100-1] = '\0'; 
        destLen = strlen(dest);
        for (i = 0; i < destLen; i++)
        {
            dest[i] = data[i];
        }
        dest[100-1] = '\0';
        printLine(dest);
        delete [] data;
    }
}
