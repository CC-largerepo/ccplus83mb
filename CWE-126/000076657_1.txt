

#include <wchar.h>

namespace GenNS1
{

typedef struct _structType
{
    char * structFirst;
} structType;



void func1(structType myStruct)
{
    char * data = myStruct.structFirst;
    {
        char dest[100];
        memset(dest, 'C', 100-1);
        dest[100-1] = '\0'; 
        memmove(dest, data, strlen(dest)*sizeof(char));
        dest[100-1] = '\0';
        printLine(dest);
        delete [] data;
    }
}

} 

#include <wchar.h>

namespace GenNS1
{

typedef struct _structType
{
    char * structFirst;
} structType;



void func1(structType myStruct);

void func3()
{
    char * data;
    structType myStruct;
    data = NULL;
    data = new char[50];
    memset(data, 'A', 50-1); 
    data[50-1] = '\0'; 
    myStruct.structFirst = data;
    func1(myStruct);
}

} 



using namespace GenNS1; 

int main(int argc, char * argv[])
{
    srand( (unsigned)time(NULL) );

    printLine("Calling ...");
    func3();
    printLine("Finished");
    return 0;
}
