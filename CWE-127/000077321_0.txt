{
public:
    virtual void action(char * data) = 0;
};



class GenClass3 : public GenClass1
{
public:
    void action(char * data);
};

}



namespace GenNS1
{

void GenClass3::action(char * data)
{
    {
        size_t i;
        char dest[100];
        memset(dest, 'C', 100-1); 
        dest[100-1] = '\0'; 
        for (i = 0; i < 100; i++)
        {
            dest[i] = data[i];
        }
        dest[100-1] = '\0';
        printLine(dest);
    }
}

}


namespace GenNS1
{



static void func3()
{
    char * data;

    data = dataBuffer;
    GenClass1* baseObject = new GenClass3;
    baseObject->action(data);

}

void func4()

    func3();
}

}
