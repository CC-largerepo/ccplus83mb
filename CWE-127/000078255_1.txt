#include <wchar.h>

namespace GenNS1
{



void func1(char * data);

void func3()
{
    char * data;
    void (*funcPtr) (char *) = func1;
    data = NULL;
    {
        char * dataBuffer = new char[100];
        memset(dataBuffer, 'A', 100-1);
        dataBuffer[100-1] = '\0';
        data = dataBuffer - 8;
    }
    funcPtr(data);
}

} 



using namespace GenNS1; 

int main(int argc, char * argv[])
