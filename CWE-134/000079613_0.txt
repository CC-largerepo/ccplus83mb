#endif



static void func3(char * data, ...)
{
    {
        va_list args;
        va_start(args, data);
        vprintf(data, args);
        va_end(args);
    }
}

void func4(char * * dataPtr)
{
    char * data = *dataPtr;
    func3(data, data);
}

static void func5(char * data, ...)
{
    {
        va_list args;
        va_start(args, data);
        vprintf("%s", args);
        va_end(args);
    }
}

void func6(char * * dataPtr)
{
    char * data = *dataPtr;
    func5(data, data);
}

        size_t dataLen = strlen(data);

        {
            if (fgets(data+dataLen, (int)(100-dataLen), stdin) != NULL)
            {
                dataLen = strlen(data);
                if (dataLen > 0 && data[dataLen-1] == '\n')
                {
                    data[dataLen-1] = '\0';
                }
            }
            else
            {
                printLine("fgets() failed");
