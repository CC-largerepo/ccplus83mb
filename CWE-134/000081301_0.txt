using namespace std;

namespace GenNS1
{



void func4(map<int, wchar_t *> dataMap);

static void func8()
{
    wchar_t * data;
    map<int, wchar_t *> dataMap;
    wchar_t dataBuffer[100] = L"";
    data = dataBuffer;
    wcscpy(data, L"fixedstringtest");
    dataMap[0] = data;
    dataMap[1] = data;
    dataMap[2] = data;
    func4(dataMap);
}

void func6(map<int, wchar_t *> dataMap);

static void func9()
{
    wchar_t * data;
    map<int, wchar_t *> dataMap;
    wchar_t dataBuffer[100] = L"";
    data = dataBuffer;
    {
        size_t dataLen = wcslen(data);
        wchar_t * environment = GETENV(ENV_VARIABLE);
        if (environment != NULL)
        {
            wcsncat(data+dataLen, environment, 100-dataLen-1);
        }
    }
    dataMap[0] = data;
    dataMap[1] = data;
    dataMap[2] = data;
    func6(dataMap);
}

void func10()
{
    func8();
    func9();
}
