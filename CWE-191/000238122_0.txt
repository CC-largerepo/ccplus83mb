#include "std_testcase.h"



static void func2()
{
    char data;
    data = ' ';
    while(1)
    {
        data = (char)RAND32();
        break;
    }
    while(1)
    {
        if (data > CHAR_MIN)
        {
            --data;
            char result = data;
            printHexCharLine(result);
        }
        else
        {
            printLine("data value is too large to perform arithmetic safely.");
        }
        break;
    }
}

static void func3()
{
    char data;
    data = ' ';
    while(1)
    {
        data = -2;
        break;
    }
    while(1)
    {
        {
            --data;
            char result = data;
            printHexCharLine(result);
        }
        break;
    }
}

void func4()
{
    func2();
    func3();
}



int main(int argc, char * argv[])
{
