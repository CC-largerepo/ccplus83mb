



void func1(short data)
{
    {
        char source[100];
        char dest[100] = "";
        memset(source, 'A', 100-1);
        source[100-1] = '\0';
        if (data < 100)
        {
            memmove(dest, source, data);
            dest[data] = '\0'; 
        }
        printLine(dest);
    }
}





void func1(short data);

void func3()
{
    short data;
    void (*funcPtr) (short) = func1;
    data = 0;
    fscanf (stdin, "%hd", &data);
    funcPtr(data);
}



int main(int argc, char * argv[])
