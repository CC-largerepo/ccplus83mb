static void func1(int data)
{
    if (data < 100)
    {

        if (dataBuffer == NULL) {exit(-1);}
        memset(dataBuffer, 'A', data-1);
        dataBuffer[data-1] = '\0';
        printLine(dataBuffer);
        free(dataBuffer);
    }
}

void func2()
{
    int data;
    void (*funcPtr) (int) = func1;

}



int main(int argc, char * argv[])
