static void func2()

var1:

        HANDLE hFile;

        hFile = CreateFileW(

                    GENERIC_READ,
                    FILE_SHARE_READ,
                    NULL,
                    CREATE_NEW,
                    FILE_ATTRIBUTE_NORMAL,
                    NULL);

        {
