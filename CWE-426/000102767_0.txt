#ifdef _WIN32
#define var1 L"cmd.exe /c dir"

#define POPEN _wpopen

#else

#define PCLOSE pclose
#endif

static int staticReturnsTrue()
{
    return 1;
}

static int staticReturnsFalse()
{
    return 0;
}

{
    wchar_t * data;
    wchar_t dataBuffer[100] = L"";
    data = dataBuffer;
    if(staticReturnsFalse())
    {

    }
    else
    {
        wcscpy(data, var2);
    }
    {
        FILE *pipe;
        pipe = POPEN(data, L"wb");
        if (pipe != NULL)
        {
            PCLOSE(pipe);
        }
    }
}

{

void func6()
