#include <wchar.h>

static int staticReturnsTrue()
{
    return 1;
}

static int staticReturnsFalse()
{
    return 0;
}



void func3()
{
    wchar_t data;
    if(staticReturnsTrue())
    {
        data = L'W';
    }
    if(staticReturnsTrue())
    {
        data = L'Z';
        printf("%02lx\n", data);
    }
}



int main(int argc, char * argv[])
{
