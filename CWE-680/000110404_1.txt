{

};



class GenClass2 : public GenClass1
{
public:
    void action(int data) const;
};

}



namespace GenNS1
{

void GenClass2::action(int data) const
{
    {
        size_t i;
        int *intPointer;
        intPointer = (int*)malloc(data * sizeof(int));
        for (i = 0; i < (size_t)data; i++)
        {
            intPointer[i] = 0; 
        }
        printIntLine(intPointer[0]);
        free(intPointer);
    }
}

}
