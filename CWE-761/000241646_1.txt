

#include <wchar.h>

namespace GenNS1
{

class GenClass1
{
public:
    virtual void action(wchar_t * data) = 0;
};



class GenClass2 : public GenClass1
{
public:
    void action(wchar_t * data);
};

}



#define SEARCH_CHAR L'S'

namespace GenNS1
{

void GenClass2::action(wchar_t * data)
{
    for (; *data != L'\0'; data++)
    {
        if (*data == SEARCH_CHAR)
        {
            printLine("We have a match!");
            break;
        }
    }
    free(data);
}

}
