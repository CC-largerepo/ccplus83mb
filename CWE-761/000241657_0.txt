#include <wchar.h>

#define var1 L"Fixed String"

static void func2()
{

    if (data == NULL) {exit(-1);}
    data[0] = L'\0';
    wcscpy(data, var1);
    if(GLOBAL_CONST_FALSE)
    {
        printLine("Benign, fixed string");
    }
    else
    {
        {
            size_t i;
            for (i=0; i < wcslen(data); i++)

                }

        }

}
