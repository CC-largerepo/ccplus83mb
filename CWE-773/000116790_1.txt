

#include <windows.h>



void func1(HANDLE * dataPtr)
{
    HANDLE data = *dataPtr;
    data = CreateFile(",
                      (GENERIC_WRITE|GENERIC_READ),
                      0,
                      NULL,
                      OPEN_ALWAYS,
                      FILE_ATTRIBUTE_NORMAL,
                      NULL);
    if (data != INVALID_HANDLE_VALUE)
    {
        CloseHandle(data);
    }
}



#include <windows.h>



void func1(HANDLE * dataPtr);

void func3()
{
    HANDLE data;
    data = INVALID_HANDLE_VALUE;
    data = CreateFile(",
                      (GENERIC_WRITE|GENERIC_READ),
                      0,
                      NULL,
                      OPEN_ALWAYS,
                      FILE_ATTRIBUTE_NORMAL,
                      NULL);
    func1(&data);
}



int main(int argc, char * argv[])
{
    srand( (unsigned)time(NULL) );

    printLine("Calling ...");
    func3();
    printLine("Finished");
    return 0;
}
